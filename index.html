from pyscript import Element def funzioncina(orario_inizio: float, orario_fine: float) -> str: paga = (orario_fine - orario_inizio) * 7 return f'{paga:.2f} euro' def calcola_guadagno(event): orario_inizio_element = Element("orario-inizio") orario_fine_element = Element("orario-fine") risultato_element = Element("risultato") orario_inizio = float(orario_inizio_element.element.value) orario_fine = float(orario_fine_element.element.value) risultato = funzioncina(orario_inizio, orario_fine) risultato_element.write(risultato) Element("calcola-btn").element.onclick = calcola_guadagno
